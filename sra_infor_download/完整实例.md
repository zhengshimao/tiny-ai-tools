## 示例

```sh
mkdir -p all

# 如果是多个SRR编号，将SRR编号放入 srr.txt

# 抓取SRA信息
cat srr.txt |while read i; do python run_get_bioinfor_from_NCBI_SRA.py -i $i -o all;done

cat all/*_download_axel.sh |grep '^axel' |grep -v 'lite' > run_axel.sh

cat all/*_download_md5.txt |grep -v '.lite' > MD5.txt

# 下载
bash run_axel.sh # 下载的文件没有后缀，与SRR编号相同

# 拆分方法1：适合常规大小PE测序
cat srr.txt |while read i
do
echo "mkdir -p ${i}_fq; fasterq-dump --threads 16 --split-3 -O ./${i}_fq ./$i  && pigz -p 16 ./${i}_fq/*.fastq"
done > run_split_v1.sh

# 拆分方法2：适合单细胞等PE测序
cat srr.txt |while read i
do
echo "mkdir -p ${i}_fq;parallel-fastq-dump --threads 16 --outdir ./${i}_fq --split-3  --gzip -s $i"
done > run_split_v2.sh

```
